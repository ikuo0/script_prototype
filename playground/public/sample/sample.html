変数、加算、比較、分岐 


    MAIN
      DATA a 1
      DATA b 2
      DATA c 0
      MATH_ADD c a b
      PRINT "c=" "@c"

      DATA flag 0
      CMP_EQUAL flag c 0
      IF flag
        PRINT "c is 0"
      ELSE
        PRINT "c is not 0"
      END_IF
    END_MAIN

関数 


    FUNCTION SUM3 a b c result
      MATH_ADD result a b
      MATH_ADD result result c
      RETURN
    END_FUNCTION

    MAIN
      DATA a 1
      DATA b 2
      DATA c 3
      DATA result 0
      CALL_FUNCTION SUM3 a b c result
      PRINT "result=" "@result"
    END_MAIN
